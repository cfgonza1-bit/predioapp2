function switchTab(el, tabId) {

  // Desactivar tabs

  el.closest('.tabs') ?

    el.closest('.tabs')

      .querySelectorAll('.tab')

      .forEach(t => t.classList.remove('active')) :

    document.querySelectorAll('.tab')

  ['tab-ingresos','tab-costos','tab-ua'].forEach(id => {

}

function calcUA() {

  const REF_UA = 450;

  const categorias = [

    { inputs: ['toro'], id: 'ua-toro' },

    { inputs: ['vaca'], id: 'ua-vaca' },

    { inputs: ['novillo'], id: 'ua-novillo' },

    { inputs: ['novilla'], id: 'ua-novilla' },

    { inputs: ['ternero'], id: 'ua-ternero' },

    { inputs: ['ternera'], id: 'ua-ternera' },

  ];

  let totalUA = 0;

  document.querySelectorAll('.ua-table tbody tr:not(.ua-total)')

    .forEach(row => {

      const inputs = row.querySelectorAll('input');

      const n    = parseFloat(inputs[0]?.value || 0);

      const peso = parseFloat(inputs[1]?.value || 0);

      const ua   = (n * peso / REF_UA);

      totalUA += ua;

      const td = row.querySelector('td:last-child');

      if (td) td.textContent = ua.toFixed(1);

    });

  const totEl = document.getElementById('ua-total-bovina');

  if (totEl) totEl.textContent = totalUA.toFixed(1);

  calcCarga(totalUA);

}


function calcCarga(totalUA) {

  const haInput = document.querySelector(

    '#tab-ua input[type="number"]');

  if (!haInput) return;

  const ha = parseFloat(haInput.value || 1);

  const ua = totalUA ||

    parseFloat(

      document.getElementById('ua-total-bovina')

        ?.textContent || 0);

  const carga = (ua / ha).toFixed(2);

  const el = document.getElementById('carga-animal');

  if (el) {

    el.textContent = `${carga} UA/ha`;

    const sub = el.nextElementSibling;

    if (sub) sub.textContent =

      `Total UA: ${ua.toFixed(1)} ÷ ${ha} ha`;

  }

}
